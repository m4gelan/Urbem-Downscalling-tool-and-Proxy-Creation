\section{Overview}
% ==============================================================================

This thesis uses \textbf{BibTeX} with the \texttt{natbib} package and the
\texttt{agsm} bibliography style, which produces Harvard-style author--year
citations. This is standard in the atmospheric science and air quality
literature.

All references are stored in \texttt{bibliography/references.bib}. This file is
pre-populated with the core references for the project. Add new entries as you
read papers.

\section{Citation Commands}
% ==============================================================================

\subsection{Basic citations}

\begin{tabular}{lll}
  \textbf{Command} & \textbf{Output} & \textbf{Use when} \\
  \hline
  \texttt{\textbackslash citet\{key\}} & Ramacher et al.\ (2021) & Part of the sentence \\
  \texttt{\textbackslash citep\{key\}} & (Ramacher et al., 2021) & Parenthetical \\
  \texttt{\textbackslash cite\{key\}} & Ramacher et al., 2021 & Avoid (use \texttt{citet}/\texttt{citep}) \\
\end{tabular}

\subsection{Multiple citations}

\begin{lstlisting}
\citep{Ramacher2021, Kuenen2022}
% Output: (Ramacher et al., 2021; Kuenen et al., 2022)
\end{lstlisting}

\subsection{Page numbers and specific parts}

\begin{lstlisting}
\citep[p.~5]{Ramacher2021}       % (Ramacher et al., 2021, p. 5)
\citep[see also][]{Kuenen2022}   % (see also Kuenen et al., 2022)
\end{lstlisting}

\section{Entry Types Used in This Project}
% ==============================================================================

\subsection{\texttt{@article} --- Journal paper (most common)}

\begin{lstlisting}
@article{LastnameYYYY,
  author  = {Lastname, Firstname and Lastname2, Firstname2},
  title   = {Title of the paper},
  journal = {Name of the Journal},
  year    = {2021},
  volume  = {12},
  number  = {11},     % issue number (optional)
  pages   = {1404},
  doi     = {10.3390/atmos12111404}
}
\end{lstlisting}

\textbf{Key:} Use \texttt{LastnameYYYY} as the citation key (e.g.,
\texttt{Ramacher2021}). If two papers share the same key, append a letter:
\texttt{Kuenen2022a}, \texttt{Kuenen2022b}.

\subsection{\texttt{@techreport} --- Technical / guidance document}

\begin{lstlisting}
@techreport{Author_ShortTitleYYYY,
  author      = {Author, Firstname},
  title       = {Full title of the document},
  institution = {Name of issuing institution},
  year        = {2025},
  type        = {Service Tool},   % optional label
  number      = {ST15},           % report number
  url         = {https://...}
}
\end{lstlisting}

\subsection{\texttt{@online} --- Website / web resource}

\begin{lstlisting}
@online{AuthorYYYY,
  author  = {{Organisation Name}},   % use double braces for corporate authors
  title   = {Page title},
  year    = {2024},
  url     = {https://example.org},
  urldate = {2026-03-08}             % date you last accessed it
}
\end{lstlisting}

\subsection{\texttt{@misc} --- Conference presentations, datasets, software}

\begin{lstlisting}
@misc{AuthorYYYY,
  author       = {Author, Firstname},
  title        = {Title of the item},
  year         = {2023},
  howpublished = {Presented at EGU General Assembly 2023, Vienna},
  note         = {Abstract EGU23-XXXX}
}
\end{lstlisting}

\section{Getting BibTeX Entries}
% ==============================================================================

\subsection{From a DOI}

Most efficient method. Use the DOI-to-BibTeX converter at:
\begin{itemize}
  \item \url{https://doi2bib.org} --- paste the DOI, get the \texttt{.bib} entry.
  \item \url{https://www.bibme.org} --- manual entry with formatting.
\end{itemize}

\subsection{From Google Scholar}

\begin{enumerate}
  \item Search for the paper on \href{https://scholar.google.com}{Google Scholar}.
  \item Click the \textbf{cite} icon (\textquotedblleft~\textquotedblright) below
        the result.
  \item Select \textbf{BibTeX} at the bottom of the popup.
  \item Copy the entry and paste it into \texttt{references.bib}.
  \item \textbf{Always verify:} Google Scholar entries often have formatting
        errors. Check author names, title capitalisation, and journal name.
\end{enumerate}

\subsection{From Zotero (recommended for managing large bibliographies)}

\begin{enumerate}
  \item Add the paper to Zotero (browser plugin or manual entry).
  \item Select the items, right-click → \textbf{Export Items} → \textbf{BibTeX}.
  \item Paste the exported entries into \texttt{references.bib}.
\end{enumerate}

\section{Common Formatting Rules}
% ==============================================================================

\subsection{Title capitalisation}
BibTeX/agsm only capitalises the first word of a title. To force capitals,
wrap the word in double braces:

\begin{lstlisting}
title = {The {UrbEm} Hybrid Method to Derive High-Resolution Emissions}
% Without braces: "The urbem hybrid method..."
% With braces:    "The UrbEm hybrid method..."
\end{lstlisting}

Always protect acronyms, proper names, and chemical species:
\texttt{\{CAMS-REG\}}, \texttt{\{Europe\}}, \texttt{\{PM2.5\}},
\texttt{\{NO\$\_x\$\}}.

\subsection{Author names}
Use \texttt{Lastname, Firstname} format, separated by \texttt{and}:
\begin{lstlisting}
author = {Ramacher, Martin Otto Paul and Kakouri, Anastasia and ...}
\end{lstlisting}
For corporate authors (organisations), use double braces to prevent BibTeX from
treating the name as ``Firstname Lastname'':
\begin{lstlisting}
author = {{European Environment Agency}}
\end{lstlisting}

\subsection{Special characters}
BibTeX uses ASCII. Replace special characters with LaTeX commands:
\begin{tabular}{ll}
  Character & LaTeX \\
  \hline
  \"a (ä) & \texttt{\textbackslash "\{a\}} \\
  \'{e} (é) & \texttt{\textbackslash '\{e\}} \\
  \v{c} (č) & \texttt{\textbackslash v\{c\}} \\
  \~{n} (ñ) & \texttt{\textbackslash \textasciitilde\{n\}} \\
\end{tabular}

\section{Compiling the Bibliography}
% ==============================================================================

On Overleaf, the bibliography compiles automatically. Locally:

\begin{lstlisting}[language=bash]
pdflatex main.tex
bibtex main
pdflatex main.tex
pdflatex main.tex
\end{lstlisting}

Or with \texttt{latexmk} (handles all passes automatically):
\begin{lstlisting}[language=bash]
latexmk -pdf main.tex
\end{lstlisting}

\section{Quick Checklist for Each New Reference}
% ==============================================================================

\begin{itemize}
  \item[$\square$] Entry type correct (\texttt{@article}, \texttt{@techreport},
        \texttt{@online})?
  \item[$\square$] Citation key follows \texttt{LastnameYYYY} convention?
  \item[$\square$] All author names present and formatted correctly?
  \item[$\square$] Acronyms and proper names protected with double braces
        in the title?
  \item[$\square$] DOI or URL included?
  \item[$\square$] Year field is a 4-digit number (not ``2021a'')?
  \item[$\square$] No duplicate citation keys in \texttt{references.bib}?
\end{itemize}
